IEnumerable<UserModel> User = null;
            using (var client = new HttpClient())
            {
                client.BaseAddress = new Uri("https://localhost:44347/api/usermodels/");
                //HTTP GET
                var responseTask = client.GetAsync("UserModel");
                responseTask.Wait();

                var result = responseTask.Result;
                if (result.IsSuccessStatusCode)
                {
                    var readTask = result.Content.ReadAsAsync<UserModel>();
                    readTask.Wait();

                    //User = readTask.Result;
                }
                else //web api sent error response 
                {
                    //log response status here..

                    User = Enumerable.Empty<UserModel>();

                    ModelState.AddModelError(string.Empty, "Server error. Please contact administrator.");
                }
            }
            return View("userpage");
